# 圆形 Circle

Circle 可绘制圆形

不设置 lineWidth 或 color 圆形不可见

# 属性

属性名 属性类型 默认值 说明
radius number 0 设置圆半径
lineColor number 0 设置边框颜色
lineWidth number 0 设置边框宽度
color number 设置填充色
anchorX 0-1 设置内部X坐标
anchorY 0-1 设置内部Y坐标

# 定义

const circle: gui.Circle = {
    name: "circle",
    type: guiType.Circle,
    color:0xffffff,
    lineColor: 0xff00cc,
    lineWidth: 1,
    radius:100
};

# 使用

{
id: "circle",
libId: ComponentId.circle,
}
